		dhcp动态分配ip


Router(config)#in f0/0
Router(config-if)#ip address 192.168.10.254 255.255.255.0
Router(config-if)#no shutdown 


Router(config-if)#in f0/1
Router(config-if)#ip address 192.168.20.254 255.255.255.0
Router(config-if)#no shutdown 



Router(config)#ip dhcp pool vlan10   给地址池一个名字vlan10  
Router(dhcp-config)#network  192.168.10.0 255.255.255.0 网段
Router(dhcp-config)#default-router 192.168.10.254 	网关
Router(dhcp-config)#dns-server 8.8.8.8			dns

Router(config)#ip dhcp pool vlan20
Router(dhcp-config)#network 192.168.20.0 255.255.255.0
Router(dhcp-config)#default-router 192.168.20.254
Router(dhcp-config)#dns-server 8.8.8.8

Router(config)#ip dhcp excluded-address 192.168.20.1 192.168.20.100  预留ip地址(地址排除)
